# pocet krokov
steps = 100

# priemer
d = 50

# stred
x = 320
y = 240

# polomer
r = d/2

for i in range(steps):
     # dlzka obluku na jednotkovej kruznici
     arc = i * (2*pi/steps)

     # prevod
     print (x+(cos(arc)*r), y+(sin(arc)*r))
