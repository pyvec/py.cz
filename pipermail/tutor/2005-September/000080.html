tak je zkus odecist :-)

priklad:
 >>> import time
 >>> from datetime import datetime
 >>> def strptime(date_string, format):
 >>>   return datetime(*(time.strptime(date_string, format)[0:6]))
 >>> d1 = strptime("2005-08-15 12:33:23", "%Y-%m-%d %H:%M:%S")
 >>> d2 = strptime("2005-08-20 12:33:23", "%Y-%m-%d %H:%M:%S")
 >>> print d2-d1
5 days, 0:00:00
