def posunHrace():
   xh = random.randint(0, 630)
   yh = random.randint(0, 479)
   global klavesy
   klavesy = keys_pressed()
   if "8" in klavesy:
       yh=yh+1
       move_to(Hrac_telo, xh, yh)
       sleep(1)

       sleep(1)

       xh=xh-1
       yh=yh+1
       move_to(Hrac_telo, xh, yh)
       sleep(1)
