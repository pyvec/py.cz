Tohle by melo fungovat od verze 2.2:


    class Klass (object) :

        def __new__ (klass, arg=None) :
            if arg is None :
                return None
            return object.__new__(klass, arg)

        def __init__ (self, arg) :
            self.arg = arg
        
        def __repr__ (self) :
            return "Klass(%r)" % self.arg

    # Pouziti
    k1, k2 = Klass(None), Klass(12)
    print k1, k2
