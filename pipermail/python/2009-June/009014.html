class atuple(tuple):
     def __getitem__(self, index):
         if isinstance(index, tuple):
             ret = []
             for i in index:
                 if isinstance(i, slice):
                     ret.extend(self[i])
                 else:
                     ret.append(self[i])
             return tuple(ret)
         return tuple.__getitem__(self, index)
