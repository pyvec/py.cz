Je to celkem jednoduché: metody definuje¹ se "self", ale volá¹ bez
"self" (ten se doplní sám, z objektu na kterém tu metodu volá¹):

class pokus:
    def __init__(self, rok):
        self.rok = rok

    def nejaka_funkce(self):
        return (self.rok + 9) % 28

    def jina_funkce(self):
        return ((self.nejaka_funkce() - 1) * 11) % 30

prvni = pokus(1845)
print (prvni.jina_funkce())
