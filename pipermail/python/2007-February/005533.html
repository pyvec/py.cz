# definujem classu, obsahuje jedno pole ako member
class cSet:
	arr = []

# pole, kde uchovam niekolko cSet-ov
list = []

# napln pole
for i in range ( 3 ):
	cset = cSet()
	cset.arr.append ( "lama" + str ( i ) )
	list.append(cset)

# vysledok
for i in list:
	print i.arr
