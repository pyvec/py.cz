Taky jednoduchsi sposob je:

import re
reCislo = re.compile('\d+')

eval2 = lambda xxx: eval(reCislo.sub(lambda x: str(int(x.group(0), 2)), 
xxx))

#priklady:
 >>> eval2('101 * (10 + 111)')
45
 >>> eval2('sqrt((10*10+100)/10)')
2.0
