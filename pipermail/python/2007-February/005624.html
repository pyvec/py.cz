#!/usr/bin/python
# -*- coding: utf-8 -*-

from Tkinter import *

class App:
  def __init__(self, okno):
    self.okno = okno
    self.okno.wm_title("Example")
    self.okno.resizable(0, 0)

    self.text1_ET = Entry(self.okno)
    self.text1_ET.grid(row=0, column=0, sticky=S, padx=10, pady=10)

    self.text2_ET = Entry(self.okno)
    self.text2_ET.grid(row=1, column=0, sticky=S, padx=10, pady=10)

    self.get_BT = Button(self.okno, text="Get text", command=self.get_text)
    self.get_BT.grid(row=2, column=0, sticky=S, padx=10, pady=10)

  def get_text(self):
    print self.text1_ET.get()
    print self.text2_ET.get()

def main():
  okno = Tk()
  app = App(okno)
  okno.mainloop()

if __name__ == '__main__':
  main()
