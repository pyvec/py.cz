def fa():
    s = 'funkce fa()'
    raise Exception

def fb():
    s = 'funkce fb()'
    fa()

def main():
    try:
        s = 'main'
        fb()
    except Exception, e:
        import inspect
        frames = inspect.trace()
        
        for ft in frames:
            print '-' * 70
            f = ft[0]
            print 'Funkce', ft[3]
            print 'Locals', f.f_locals

if __name__=='__main__':
    main()
