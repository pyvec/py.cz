Zdravim, potreboval bych upravit tento skript, aby zobrazoval konkretniho 
uzivatele funkce: ldapsearch -x uid=XXXX, misto skupin, dekuji za 
jakoukoli pomoc

#!/usr/bin/python
#

import ldap

url="ldap://sun.cb.madeta.cz/"
dn="dc=jihomik-group,dc=cz"

l = ldap.initialize(url)
l.bind_s("","", ldap.AUTH_SIMPLE):

for name,attrs in l.search_s(dn,ldap.SCOPE_ONELEVEL,"objectclass=*",[]):
    print "name: %s " %name
