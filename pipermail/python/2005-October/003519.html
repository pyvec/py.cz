database  = ["nic", "neco", "copak", "cop", "oko", "okov"]
inventory = ["p", "a", "c", "v", "o", "k"]
 
def scan_database(inventory):
    matched = []
    for word in database:
        find = True
        new = inventory[:]
        for char in word:
            if char not in new:
                find = False
                break
            else:
                pos = new.index(char)
                del new[pos]
        if find:
            matched.append(word)
    return matched
 
found = scan_database(inventory)
print found
