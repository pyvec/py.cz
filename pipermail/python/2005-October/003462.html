fin = file('vstup.txt')
fout = file('vystup.txt', 'w')

processed = set()     # prazdna mnozina jiz zpracovanych 

for line in fin:
    if line[0] not in '* \t':
        # Novy nazev souboru, zaciname sbirat dalsi sekci.
        fout.write(line)
        processed.clear()

    else:
        # Sbirame unikatni radky sekce.
        if line not in processed:
            processed.add(line)
            fout.write(line)

fin.close()
fout.close()
