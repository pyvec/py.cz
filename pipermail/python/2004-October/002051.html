Ja znam tyto druhy promennych:

automaticke - jejich platnost je v ramci vykonavane funkce/metody. 
	a=123

globalni - platnost v ramci modulu 
	global a
	a=123

promenne instance tridy
class zz:
	def nastav(s):
		s.a=123
	
	def vypis(s):
		print s.a
	
promenne tridy
class zz:
