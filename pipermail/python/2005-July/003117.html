Dulezite je pouzit raw-unicode-escape misto unicode-escape ;-)

#!/usr/bin/python
# -*- coding: utf-8 -*-

import sys

try:
  fr = open("iso-8859-2", "r")
  fw = open("unicode-escape", "w")
except IOError, chyba:
  print chyba
  sys.exit(1)

for radek in fr.readlines():
  radek = radek.decode('iso-8859-2').encode('raw-unicode-escape')
  fw.write(radek)

fr.close()
fw.close()
