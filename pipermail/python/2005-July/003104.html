from Tkinter import *

OPTIONS = [
    "egg",
    "bunny",
    "chicken"
]

root = Tk()

var = StringVar()
var.set(OPTIONS[2]) # default value

w = OptionMenu (root, var, *OPTIONS)
w.pack()

root.mainloop()
