Just having fun.

    $ python
    >>> a = {}
    >>> b = {'a': a}
    >>> a['b'] = b
    >>> a
    {'b': {'a': {...}}}
    >>> b
    {'a': {'b': {...}}}
    >>> import cPickle
    >>> ap = cPickle.dumps(a)
    >>> del a, b
    >>> len(ap)
    26
    >>> a = cPickle.loads(ap)
    >>> a
    {'b': {'a': {...}}}
    >>> a['b']
    {'a': {'b': {...}}}
    >>> a['b']['a']
    {'b': {'a': {...}}}
    >>> a['b']['a']['b']
    {'a': {'b': {...}}}
    >>>
