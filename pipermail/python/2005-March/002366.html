import httplib, os

www="www.geon.wz.cz"
souborNet="/zip/livewires.zip"

# ulozeni na local

f=file(souborLocal,'wb')      # otevri soubor f pro BINARNI zapis
f.write(data)
f.close()

print "Soubor", souborLocal, "ulozen do", os.getcwd()
