class TestClass(object):
    def __init__(self):
        self.var1 = 1
        self.var2 = '2'

    def func1(self):
        self.var3 = 3

    def func2(self):
        self.var4 = '4'

# utility func
def get_vars(inst):
    res = []
    for attr_name in inst.__dict__:
        attr = getattr(inst, attr_name)
        if not attr_name.startswith('_') and not callable(attr):
            res.append(attr_name)
    return res

if __name__ == '__main__':
    x = TestClass()
    print get_vars(x)
    x.func1()
    print get_vars(x)
    x.func2()
    print get_vars(x)
