class Master:
	def a(self, x=0):
		print 'jeden argument'

class Commander(Master):
	def a(self, x=0,y=0):
		print 'dva argumenty'
		
	def b(self, x=0, y=0):
		Master().a()
		

a = Master()

b = Commander()

b.b()	# puvodni metoda a
