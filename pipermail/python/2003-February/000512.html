zkuste nasledujici kod:

***************** glob.py
def f():
    print "g"

globVar=["set from glob"]

***************** fun1.py
from g import *

def ff():
    print "f"

def fun1():
   global globVar
   global f
   globVar[0]="set from fun1()"
   f = ff

****************** main.py
from g import *
from fun1 import *

def fm():
    print "m"

def main():
   global globVar
   global f
   globVar[0]='set from main()'
   f=fm

print globVar
f()
main()

f()
fun1()
