from itertools import groupby


def summarize(data):
    get_b = lambda x: x['b']
    get_a = lambda x: x['a']
    return (
        {'a': val, 'b': sum(map(get_b, group))}
        for (val, group)
        in groupby(data, key=get_a)
    )

data = [{'a': 4, 'b': 3}, {'a': 4, 'b': 5}, {'a': 1, 'b': 3}, {'a': 3, 'b':
2}]
print list(summarize(data))  # [{'a': 4, 'b': 8}, {'a': 1, 'b': 3}, {'a':
3, 'b': 2}]
