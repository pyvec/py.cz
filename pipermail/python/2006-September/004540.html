def foo(text, treshold):
     result = []
     buff = []
     for word in text.split():
         if len(' '.join(buff + [word])) > treshold:
             result.append(' '.join(buff))
             buff = [word]
         else:
             buff.append(word)
     result.append(' '.join(buff))
     return result
