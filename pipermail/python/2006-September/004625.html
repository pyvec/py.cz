class Timer(threading.Thread):
 def __init__(self,sleep,port,action):
   self.sleep=sleep
   self.port=port
   self.action=action
   threading.Thread.__init__(self)

 def run(self):
   while 1:
     time.sleep(self.sleep)
     buff=self.port.inWaiting()
     if buff>0:
       data=self.port.read(buff)
       self.action(data)
