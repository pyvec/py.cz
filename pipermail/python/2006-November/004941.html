class Trida1():
  def self.akce:
    nejaka akce...
    ...
class Trida2():
  def self.akce:
    jina akce...
    ....

a = "Trida1"

a.akce /* by se pak interpretovalo jako Trida1.akce.*/

a = "Trida2"

a.akce  /* by se pak interpretovalo jako Trida1.akce. */
