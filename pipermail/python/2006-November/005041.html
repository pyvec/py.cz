Zkusil jsem super a vysledek je ten samy.
class a(object):
    def prn(self):
        print 'a',self
    def ma(self):
        self.prn()
    def aa(self):
        pass

class b(a):
    def prn(self):
        print 'b',self
    def ma(self):
        '''prekryje a.ma()'''
        super(b,self).ma()
    def ba(self):
        pass
