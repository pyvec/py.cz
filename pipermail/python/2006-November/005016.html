class linear():
    def __init__(self,c):
        self.c=c
    def __call__(self,x):
        return self.c * x

f1=linear(1)
f2=linear(5)
for x in range(5):
    print "f1(%d)=%d, f2(%d)=%d" % (x,f1(x),x,f2(x))
