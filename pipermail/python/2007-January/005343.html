import timeit

def f():
   a = [(1,2),(3,4)]
   for i in range(len(a)):
       a[i] = list(a[i])
   

t=timeit.Timer("f()","from __main__ import f")
print "f():", t.timeit(opak)


def g():
   a = [ (1, 2), (3, 4) ]
   for i, elem in enumerate(a):
       a[i] = list(elem)

t=timeit.Timer("g()","from __main__ import g")
print "g():", t.timeit(opak)
