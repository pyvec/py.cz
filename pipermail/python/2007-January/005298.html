Zdravim,

ja pouzivam tento zpusob:
-------------------------

title = "muj web"

menu_psp = psp.PSP(req, "menu_tmpl.html", vars = {})

content_psp = psp.PSP(req, "content_tmpl.html",
			vars = {'obsah': "nejaky obsah"})

web = psp.PSP(req, "web_tmpl.html", vars={'menu_tmpl':menu_psp,
			'content':content_psp,
                        'title':title})
